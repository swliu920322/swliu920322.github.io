闭包在vue3的使用

  • 早期做项目，基于简单这一个出发点，所谓的封装也只是一些执行顺序的问题。
    对闭包用的也少。
  • 最近看了一个开源vue3后台的处理方式，可能这就是闭包该有的写法吧。
    本质是闭包内部维持状态，暴露闭包set函数的集合和内部状态的set，通过vue的emit传回到组件本身。
    把属性合并作为v-bind的对象体，使用者只关心函数调用即可。
  • 封装本身用来控制抽象的内容，分可预测和难预测的部分，通常解决掉是重复的可预测内容。
    之前简单的封装一部分是对于模板的，wrapper函数，一部分是逻辑的logic函数，之前是做了2份的，本身还得配合使用。
    现在wrapper函数和模板通过内部闭包状态合并了，对于使用者来说是一个很舒服的状态。

vue3配合闭包目前的封装思想

真实组件使用

  const app = vue.createApp({
  template: `<BasicTable @register="register" />`,
  setup() {
    const [register, methods] = useTable({
      loading: false,
    });
    return {
      register,
      setLoading() {
        console.log(methods.setLoading(false));
      },
    };
  },
});

组件内部封装

  function useLoading(initialProps) {
    const loadingRef = ref(unref(initialProps).loading);
    watch(
      // 监听props之后的变化，更新状态
      () => unref(initialProps).loading,
      (val) => (loadingRef.value = val)
    );
    return {
      getLoadingRef: computed(() => unref(loadingRef)),
      setLoading: (value) => (loadingRef.value = value),
    };
  }

  const Table = vue.createApp({
  template: `<ATable ref="tableRef" v-bind="getBindValue" />`,
  setup(props, { attrs, emit }) {
    const tableRef = ref();
    const innerPropsRef = ref({}); // 始终获取最新的props
    const getProps = computed(() => ({ ...props, ...unref(innerPropsRef) }));
    const { getLoadingRef, setLoading } = useLoading(getProps);
    const setInnerPropRef = (value) => {
      innerPropsRef.value = { ...unref(innerPropsRef), ...value };
    }; // 使用最新的状态
    const getBindValue = computed(() => {
      return { ...props, ...attrs, ...unref(getProps), loading: unref(getLoadingRef) };
    });
    const tableAction = { setLoading, setInnerPropRef };
    // 把闭包函数集合emit为事件参数
    emit("register", tableAction);
    return { tableRef, getBindValue };
  },
});

useTable内部

function useTable(props) {
  let instance = null;
  function register(actions) {
    actions.setInnerPropRef(props);
    instance = actions;
  }
  function getInstance() {
    if (!instance) {
      throw new Error("instance has not ready");
    }
    return instance;
  }
  const otherMethods = {
    setLoading: (val) => getInstance()?.setLoading(val),
  };
  return [register, otherMethods];
}

组件内部封装

function useLoading(initial) {
  // 假设的loading响应式函数
  let loading = initial;
  return {
    getLoading: () => loading,
    setLoading: (val) => (loading = val),
  };
} // 一个假设的状态响应式函数 同上
function useState(initial) {
  let state = initial;
  return {
    getState: () => state,
    setState: (val) => (state = val),
  };
} // 真实table组件内部状态
function Table() {
  const { getLoading, setLoading } = useLoading();
  const { getState, setState } = useState();
  const getComputedLoading = () => getLoading();
  const getComputedState = () => getState();
  const setMethods = {
    setState,
    setLoading,
    getComputedLoading,
    getComputedState,
    setProps(props) {
      setLoading(props?.loading);
      setState(props?.state);
    },
  }; // emit的东西暂时存起来
  actions = setMethods;
  return {
    getMethods: { getComputedLoading, getComputedState },
    setMethods,
  };
}

useTable内部

function useTable(props) {
  // 解析业务逻辑的核心
  let instance = null;
  const register = (actions) => {
    instance = actions;
    actions.setProps(props);
  };
  function getInstance() {
    if (!instance) throw new Error("instance has not init!");
    return instance;
  }
  const methods = {
    setLoading: (val) => getInstance().setLoading(val),
    getLoading: () => getInstance().getComputedLoading(),
    getState: () => getInstance().getComputedState(),
    setState: (val) => getInstance().setState(val),
  };
  return [register, methods];
}
